Interfaces


La interfaz. Se construye substituyendo la clase por la interfaz:


interface Instrumento {
  int i = 5;
  void tocar();
  String queInstrumento();
  void afinar();
}

class Viento implements Instrumento {
  public void tocar() {
    System.out.println("Viento.tocar()");
  }

  public String queInstrumento() { 
    return "Viento";
  }
  public void afinar() { 
  }
}

class Percusion implements Instrumento {
  public void tocar() {
    System.out.println("Percusion.tocar()");
  }
  public String queInstrumento() { 
    return "Percusion";
  }

  public void afinar() {
  }
}

class Cuerda implements Instrumento {
    public void tocar() {
      System.out.println("Cuerda.tocar()");
    }

    public String queInstrumento() { 
      return "Cuerda";
    }

    public void afinar() {
    }
    
}

public class MusicaDriver {
  
  static void armonizar(Instrumento i) {
    
    i.tocar();
  }
  static void armonizarOrquesta(Instrumento[] or) {

      armonizar(or[i]);
  }
  public static void main(String[] args) {
    Instrumento[] orquesta = new Instrumento[5];
    int i = 0;

    orquesta[i++] = new Percusion();
    orquesta[i++] = new Cuerda();
    orquesta[i++] = new Metal();
    orquesta[i++] = new Madera();
    armonizarOrquesta(orquesta);
  }
}
